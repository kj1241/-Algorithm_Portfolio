#include <iostream>
#include <Windows.h>
using namespace std;

typedef int ElemenTpye;




typedef struct tag_node
{
	ElemenTpye data;
	struct tag_node* left;
	struct tag_node* right;
	struct tag_node* parent;
}TreeNode;

void tree_print(const TreeNode *node)
{
	if(!node) return;

	tree_print(node->left);
	printf("%c  ",node->data);
	tree_print (node->right);
}


int main()
{
	TreeNode plus, mul,minus, int1,int2,int3, int4;

	plus.data='+'; plus.parent=NULL; plus.left=&mul; plus.right=&minus;
	mul.data='*'; mul.parent=&plus; mul.left=&int1;mul.right=&int2;
	minus.data='-';minus.parent=&plus; minus.left=&int3;minus.right=&int4;
	int1.data='1';int1.parent=&mul; int1.left=NULL; int1.right=NULL;
	int2.data='2';int2.parent=&mul; int2.left=NULL; int2.right=NULL;
	int3.data='7';int3.parent=&minus; int3.left=NULL; int3.right=NULL;
	int4.data='8';int4.parent=&minus; int4.left=NULL; int4.right=NULL;

	tree_print(&plus);

	return 0;
}
